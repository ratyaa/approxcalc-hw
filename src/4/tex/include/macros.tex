% svg inclusion
\newcommand{\svg}[3][0.7] {
  \begin{figure}[h]
    \begin{center}
      \fontsize{7}{8}\selectfont
      \includesvg[scale=#1]{\picdir #2}
      \caption{#3}
    \end{center}
  \end{figure}
}

  \end{align#1}
}

\newcommand{\feq}[2][*] {
  \begin{equation#1}
    #2
  \end{equation#1}
}
